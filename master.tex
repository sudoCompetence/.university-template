% Section: Definitions Examples 
%-------------------------------------------------------------------------------
\section{Definitions}\index{Introductions!Definitions}

\begin{definition}[Definition name]
    Given a vector space $E$, a norm on $E$ is an application, denoted
    $||\cdot||$, $E$ in $\mathbb{R}^+=[0,+\infty[$ such that:
    \begin{align}
        & ||\mathbf{x}||=0\ \Rightarrow\ \mathbf{x}=\mathbf{0}\\
        & ||\lambda \mathbf{x}||=|\lambda|\cdot ||\mathbf{x}||\\
        & ||\mathbf{x}+\mathbf{y}||\leq ||\mathbf{x}||+||\mathbf{y}||
    \end{align}
\end{definition}

\begin{example}
    Let $G=\{x\in\mathbb{R}^2:|x|<3\}$ and denoted by: $x^0=(1,1)$:
    \begin{equation}
        f(x)=\left\{\begin{aligned} & \mathrm{e}^{|x|} & & \text{si $|x-x^0|\leq
            1/2$}\\ & 0 & & \text{si $|x-x^0|> 1/2$}\end{aligned}\right.
    \end{equation}
    The function $f$ has bounded support, we can take
    $A=\{x\in\mathbb{R}^2:|x-x^0|\leq 1/2+\epsilon\}$ for all
    $\epsilon\in\mathopen{]}0\,;5/2-\sqrt{2}\mathclose{[}$.
\end{example}
